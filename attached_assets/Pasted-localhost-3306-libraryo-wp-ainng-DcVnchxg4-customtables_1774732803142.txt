-- 1. What's in the gifttype table? (looking for duplicate Trappings entries)

FROM DcVnchxg4_customtables_table_gifttype
ORDER BY ct_type_name;


-- 2. How many gift_type_map entries point to each gifttype?
SELECT gt.ct_id, gt.ct_type_name, gt.ct_slug, COUNT(m.id) AS mapped_gifts
FROM DcVnchxg4_customtables_table_gifttype gt
LEFT JOIN DcVnchxg4_customtables_table_gift_type_map m ON m.type_id = gt.ct_id
GROUP BY gt.ct_id, gt.ct_type_name, gt.ct_slug
ORDER BY gt.ct_type_name;


-- 3. What does the giftclass table look like, and how many gifts point to each?
SELECT gc.ct_id, gc.ct_class_name, gc.ct_slug, COUNT(g.ct_id) AS gift_count
FROM DcVnchxg4_customtables_table_giftclass gc
LEFT JOIN DcVnchxg4_customtables_table_gifts g ON g.ct_gift_class = gc.ct_id
GROUP BY gc.ct_id, gc.ct_class_name, gc.ct_slug
ORDER BY gc.ct_class_name;


-- 4. Which gifts have "Trappings" in their name, and what class ID are they pointing to?
SELECT g.ct_id, g.ct_gifts_name, g.ct_gift_class, gc.ct_class_name
FROM DcVnchxg4_customtables_table_gifts g
LEFT JOIN DcVnchxg4_customtables_table_giftclass gc ON gc.ct_id = g.ct_gift_class
WHERE g.ct_gifts_name LIKE '%Trappings%'
ORDER BY g.ct_gifts_name;
